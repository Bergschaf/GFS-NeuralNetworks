#+BEGIN_SRC elisp
(let ((image-files (directory-files "/path/to/your/directory" t "\\.jpg\\|\\.png\\|\\.gif$")))
  (with-current-buffer (get-buffer-create "*Images*")
    (erase-buffer)
    (org-mode)
    (dolist (file image-files)
      (insert (format "[[file:%s]]\n" file)))
    (org-display-inline-images)))
#+END_SRC
